00169     settings.setValue("imagesDirectory", imagesDirectory());
00170 
00171     settings.beginGroup("shortcuts");
00172     for (int i = 0; i < m_actions->count(); ++i) {
00173         const QTreeWidgetItem *shortcutGroup = ui->shortcutsTreeWidget->topLevelItem(i);
00174         QPair<QString, QList<QAction*> > actionGroup = m_actions->at(i);
00175         for (int j = 0; j < actionGroup.second.count(); ++j) {
00176             const QTreeWidgetItem *shortcutItem = shortcutGroup->child(j);
00177             QAction *action = actionGroup.second.at(j);
00178             action->setShortcut(shortcutItem->text(2));
