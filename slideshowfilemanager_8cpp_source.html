00053 void SlideshowFileManager::saveSlideshow(Slideshow &slideshow, const QString &directory)
00054 {
00055     if (!slideshow.hasChanged())
00056         return;
00057 
00058     QFile file(slideshow.path(directory));
00059     if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
00060         qWarning("SlideshowFileManager::saveSlideshow(Slideshow, QString): cannot open file \"%s\" (%s).",
00061                  qPrintable(file.fileName()), qPrintable(file.errorString()));
00062         return;
00063     }
00064 
00065     QXmlStreamWriter xml(&file);
00066     xml.setAutoFormatting(true);
00067     xml.writeStartDocument("1.0", true);
00068     xml.writeStartElement("slideshow");
00069     Qsm::SortFlags sort = slideshow.sortOrder();
00070     if (sort.testFlag(Qsm::Name) || sort.testFlag(Qsm::Date)) {
00071         xml.writeAttribute("sort", sort.testFlag(Qsm::Name) ? "name" : "date");
00072         if (sort.testFlag(Qsm::Reversed))
00073             xml.writeAttribute("sortOrdner", "desc");
00074     }
00075     if (!slideshow.comment().isEmpty())
00076         xml.writeAttribute("comment", slideshow.comment());
00077     foreach (const SlideshowImage &image, slideshow.images()) {
00078         xml.writeStartElement("image");
00079         xml.writeAttribute("path", QDir::toNativeSeparators(image.path()));
00080         if (image.randomFactor() != 1)
00081             xml.writeAttribute("randomFactor", QString::number(image.randomFactor()));
00082         if (!image.comment().isEmpty())
00083             xml.writeAttribute("comment", image.comment());
00084         xml.writeEndElement();
00085     }
00086     xml.writeEndElement();
00087     xml.writeEndDocument();
00088     file.close();
00089 
00090     slideshow.resetChanged();
00091 }

00114 void SlideshowFileManager::loadSlideshow(const QFileInfo &fileInfo)
00115 {
00116     QFile file(fileInfo.filePath());
00117     if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
00118         qWarning("SlideshowFileManager::loadSlideshow(QFile): cannot open file \"%s\" (%s).",
00119                  qPrintable(file.fileName()), qPrintable(file.errorString()));
00120         return;
00121     }
00122 
00123     QString slideshowName = fileInfo.fileName();
00124     slideshowName = slideshowName.left(slideshowName.length() - Slideshow::FILE_EXTENSION.length());
00125     Slideshow slideshow(slideshowName);
00126 
00127     QXmlStreamReader xml(&file);
00128     while (!xml.atEnd()) {
00129         if (xml.readNext() == QXmlStreamReader::StartElement) {
00130             const QString elementName = xml.name().toString();
00131             const QXmlStreamAttributes attr = xml.attributes();
00132             if (elementName == "slideshow") {
00133                 if (attr.hasAttribute("sort")) {
00134                     Qsm::SortFlags sortFlags = Qsm::Unsorted;
00135                     QString sort = attr.value("sort").toString();
00136                     if (sort == "name")
00137                         sortFlags = Qsm::Name;
00138                     else if (sort == "date")
00139                         sortFlags = Qsm::Date;
00140                     if (attr.hasAttribute("sortOrdner")) {
00141                         QString sortOrdner = attr.value("sortOrdner").toString();
00142                         if (sortOrdner == "desc")
00143                             sortFlags |= Qsm::Reversed;
00144                     }
00145                     slideshow.setSortOrder(sortFlags);
00146                 }
00147                 if (attr.hasAttribute("comment")) {
00148                     QString comment = attr.value("comment").toString();
00149                     slideshow.setComment(comment);
00150                 }
00151             } else if (elementName == "image") {
00152                 if (attr.hasAttribute("path")) { // mandatory attribute
00153                     QString path = attr.value("path").toString();
00154                     SlideshowImage image(path);
00155                     if (attr.hasAttribute("randomFactor")) {
00156                         QString randomFactor = attr.value("randomFactor").toString();
00157                         image.setRandomFactor(randomFactor.toInt());
00158                     }
00159                     if (attr.hasAttribute("comment")) {
00160                         QString comment = attr.value("comment").toString();
00161                         image.setComment(comment);
00162                     }
00163                     slideshow.addImage(image);
00164                 }
00165             }
00166         }
00167     }
00168     slideshow.resetChanged();
00169     if (xml.hasError()) {
00170         qWarning("SlideshowFileManager::loadSlideshow(QFile): parse error in file \"%s\" (%s at line %li and column %li).",
00171                  qPrintable(file.fileName()), qPrintable(xml.errorString()), (long)xml.lineNumber(), (long)xml.columnNumber());
00172     } else
00173         emit slideshowLoaded(slideshow);
00174     file.close();
00175 }
