00057 {
00058     GLint x = 0, y = 0;
00059     GLsizei w = width, h = height;
00060     int imageWidth, imageHeight;
00061 
00062     if (m_rotate == 90.0 || m_rotate == 270.0) {
00063         imageWidth = m_imageHeight;
00064         imageHeight = m_imageWidth;
00065     } else {
00066         imageWidth = m_imageWidth;
00067         imageHeight = m_imageHeight;

00071         double wr = (double)width / imageWidth;
00072         double hr = (double)height / imageHeight;

00075             y = (height - h) / 2;
00076         } else if (wr > hr) { // full height
00077             w = int(hr * imageWidth);
00078             x = (width - w) / 2;

00081         m_translate.setY(y);
00082         m_zoom = (double)w / imageWidth;
00083     } else {
00084         x = m_translate.x();
00085         y = m_translate.y();
00086         w = imageWidth * m_zoom;
00087         h = imageHeight * m_zoom;
00088     }
00089     m_actualImageWidth = w;
00090     m_actualImageHeight = h;
00091 
00092     glViewport(x, y, w, h);
00093 
00094     glMatrixMode(GL_PROJECTION);
00095     glLoadIdentity();
