00078 bool ImageListModel::setData(const QModelIndex &index, const QVariant &value, int role)
00079 {
00080     if (!index.isValid() || index.row() >= m_imageInfoList.count())
00081         return false;

00083     if (role == Qt::EditRole) {
00084         QFileInfo &fileInfo = m_imageInfoList[index.row()].fileInfo();

00092                 // update the file info reference

00095                 return true;

00097             qWarning("File \"%s\" could not be renamed to \"%s\"", qPrintable(oldPath), qPrintable(newPath));
